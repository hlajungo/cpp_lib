# cmake compile cheetsheet

cmake -S PROJECTNAME -B build/PROJECTNAME
cmake --build build/PROJECTNAME
cmake --install build/PROJECTNAME --prefix ./install/PROJECTNAME

cmake -S standalone/ -B build/standalone
cmake --build build/standalone
./build/standalone/PROJECTNAME_standalone
cmake --install build/standalone/ --prefix ./install/standalone

cmake -S test/ -B build/test
cmake --build build/test
ctest --test-dir ./build/test --output-on-failure

cmake -S documentation/ -B build/documentation
cmake --build build/documentation --target GenerateDocs
open build/documentation/doxygen/html/index.html

# gen ctags

ctags -R --langmap=c++:+.cu .
